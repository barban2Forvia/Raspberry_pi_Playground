# Get the exact filename (e.g. 99-ubuntu.conf)
FNAME=$(basename $(grep -rl apparmor_restrict_unprivileged_userns /usr/lib/sysctl.d/))

# Copy it to /etc/sysctl.d/ — same name takes precedence over /usr/lib/sysctl.d/
sudo cp /usr/lib/sysctl.d/$FNAME /etc/sysctl.d/$FNAME

# Override the value to 0 in your copy
sudo sed -i 's/apparmor_restrict_unprivileged_userns\s*=\s*1/apparmor_restrict_unprivileged_userns=0/' /etc/sysctl.d/$FNAME

# Apply and verify
sudo sysctl --system
cat /proc/sys/kernel/apparmor_restrict_unprivileged_userns
